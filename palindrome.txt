#include <stdio.h>

int main()
{
    int num,temp,r,sum=0;
    printf("\n enter number: ");
    scanf("%d",&num);
    temp=num;
    while(num>0)
    {
        r=num%10;
        sum=sum*10+r;
        num=num/10;
    }
    if(sum==temp)
    printf("\n palindrome");
    else
    printf("\n not a palindrome");
    return 0;
}
