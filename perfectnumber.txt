#include <stdio.h>

int main()
{
    int num, sum = 0;

    printf("Enter a number: ");
    scanf("%d", &num);

    // Loop through all numbers from 1 to num-1
    for(int i = 1; i < num; i++)
    {
        if(num % i == 0)
        {
            sum += i;  // Add divisor to sum
        }
    }

    // Check if number is perfect
    if(sum == num)
    {
        printf("%d is a perfect number.\n", num);
    }
    else
    {
        printf("%d is not a perfect number.\n", num);
    }

    return 0;
}
