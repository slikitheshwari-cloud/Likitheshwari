#include <stdio.h>

int main()
{
    int i,n,temp,j;
    printf("\n enter array size");
    scanf("%d",&n);
    int arr[n];
    printf("\n enter array elements");
    for(i=0;i<n;i++)
    {
        scanf("\n %d",&arr[i]);
    }
    printf("\n enter array elements before sorting");
    for(i=0;i<n;i++)
    {
        printf(" %d",arr[i]);
    }
    for(i=0;i<n-1;i++)
    {
        for(j=0;j<n-i-1;j++)
        {
            if(arr[j]>arr[j+1])
            {
                temp=arr[j];
                arr[j]=arr[j+1];
                arr[j+1]=temp;
            }
        }
    }
    printf("\n enter array elements after sorting");
    for(i=0;i<n;i++)
    {
        printf(" %d",arr[i]);
    }
    return 0;
}
